apt update
apt install apache2 php libapache2-mod-php php-cli php-mysql php-gd php-xml php-curl php-mbstring php-intl php-zip php-soap php-bcmath
cd /var/www/html/
wget https://download.moodle.org/download.php/direct/stable405/moodle-latest-405.tgz
tar -zxvf moodle-latest-405.tgz 
mkdir /var/www/moodledata
chown -R www-data:www-data /var/www/moodledata/
chmod -R 777 /var/www/moodledata/
chown -R www-data:www-data moodle
chmod -R 777 moodle
nano /etc/php/8.1/apache2/php.ini
cari max_input_vars lalu enable dan ubah ke 5000
cari memory_limit lalu ubah ke 512M
cari post_max_size lalu ubah ke 100M
cari upload_max_filesize lalu ubah ke 100M
lalu simpan
systemctl restart apache2
